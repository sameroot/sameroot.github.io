线上安装hadoop过程

发表于 2015-07-01

1. 检查机器是否安装jdk。如果已安装jdk，需要先把java改名

java -version

2.安装jdk

rpm -ivh jdk-7u79-linux-x64.rpm

3.安装hadoop

#添加hadoop用户

groupadd hadoop
useradd -g hadoop hadoop
echo 'my_passwd' | passwd hadoop --stdin
echo 'hadoop    ALL=(ALL)       NOPASSWD: /opt/install/hadoop/mybin/*' >> /etc/sudoers
su - hadoop -c 'ssh-keygen -t rsa'
su - hadoop -c 'cat ~/.ssh/id_rsa.pub >> ~/.ssh/authorized_keys'
su - hadoop -c 'chmod 400 ~/.ssh/authorized_keys'

#设置master到slaver的免登录

echo 'ssh-rsa xxx' >> /home/hadoop/.ssh/authorized_keys

#配置javahome环境变量

echo '' >> /home/hadoop/.bashrc
echo '#java' >> /home/hadoop/.bashrc
echo 'export JAVA_HOME=/usr/java/jdk1.7.0_79' >> /home/hadoop/.bashrc
echo 'export CLASSPATH=.:$JAVA_HOME/jre/lib/rt.jar:$JAVA_HOME/lib/dt.jar:$JAVA_HOME/lib/tools.jar' >> /home/hadoop/.bashrc
echo 'export PATH=$PATH:$JAVA_HOME/bin' >> /home/hadoop/.bashrc

#如果ssh端口号不是默认的,修改配置

echo 'export HADOOP_SSH_OPTS="-p 65001"' >> /home/hadoop/.bashrc

#初始化文件夹

mkdir -p /data/hadoop
mkdir -p /data/hadoop/tmp
mkdir -p /data/hadoop/data
mkdir -p /data/hadoop/dfs
mkdir -p /data/hadoop/dfs/name
mkdir -p /data/hadoop/dfs/data
chown -R hadoop:hadoop /data/hadoop

#安装hadoop

tar zxf hadoop-2.6.0.tar.gz
mv hadoop-2.6.0 hadoop
mv hadoop /opt/install/
chown -R hadoop:hadoop /opt/install/hadoop/

4.master修改hosts(注意ip和机器名之间只能一个空格,且这几行位于文件最上面):

10.168.75.110 JRTMv2-API-1 #hadoop
10.168.58.215 JRTMv2-API-2 #hadoop
10.168.137.6 JRTMv2-API-3 #hadoop

5.master同步hosts到slaves：

su hadoop - /opt/install/hadoop/mybin/h_synchro_host_client
#hadoop #install
